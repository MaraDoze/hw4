INSERT INTO `order` (order_id, customer_id, address_id)
SELECT 
    ROW_NUMBER() OVER () AS order_id,
    customer.customer_id,
    address.address_id
FROM
    (SELECT * FROM customer LIMIT 350) AS customer -- Limiting customer records to 350
    CROSS JOIN (SELECT * FROM address LIMIT 100) AS address -- Limiting address records to 18
LIMIT 350;
